def self.long_name (x)
  x = x.to_s # Make sure x is a string
  # Loop through the attrs until we find a match
  match = nil
  self.where({}).each do |b| 
    if b.name.downcase.include? x.downcase
      match = b.name
      break
    end
  end
  return match
end
